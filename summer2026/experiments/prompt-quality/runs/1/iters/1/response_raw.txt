@register_scheduler_init(key="scheduler_high_006")
def scheduler_high_006_init(s):
    from collections import deque

    s.tick = 0

    # Per-priority de-duplicated FIFO of pipeline_ids
    s.q_query = deque()
    s.q_interactive = deque()
    s.q_batch = deque()
    s.enqueued_query = set()
    s.enqueued_interactive = set()
    s.enqueued_batch = set()

    # Latest Pipeline object reference by id (sim may pass objects repeatedly)
    s.pipeline_by_id = {}

    # Pipeline bookkeeping
    s.pipeline_meta = {}  # pid -> dict(arrival_tick, completed_ops, failures)
    s.pipeline_permanent_fail = set()

    # Operator -> pipeline attribution (learn from results)
    s.op_to_pipeline = {}  # id(op) -> pid

    # Per-operator resource learning
    s.op_ram_est = {}  # id(op) -> ram
    s.op_cpu_est = {}  # id(op) -> cpu
    s.op_failures = {}  # id(op) -> int
    s.op_ooms = {}  # id(op) -> int
    s.op_timeouts = {}  # id(op) -> int

    # Retry / fail-fast bounds (keep high completion; only give up when clearly impossible)
    s.max_op_failures = 10
    s.max_pipeline_failures = 40

    # Initial sizing (absolute) + mild scaling with pool size
    s.min_cpu = {
        Priority.QUERY: 1.0,
        Priority.INTERACTIVE: 1.0,
        Priority.BATCH_PIPELINE: 0.5,
    }
    s.min_ram = {
        Priority.QUERY: 8.0,
        Priority.INTERACTIVE: 8.0,
        Priority.BATCH_PIPELINE: 6.0,
    }

    s.base_cpu_abs = {
        Priority.QUERY: 6.0,
        Priority.INTERACTIVE: 4.0,
        Priority.BATCH_PIPELINE: 2.0,
    }
    s.base_ram_abs = {
        Priority.QUERY: 32.0,
        Priority.INTERACTIVE: 24.0,
        Priority.BATCH_PIPELINE: 16.0,
    }

    # Fractional component (helps scale across cluster sizes) + initial caps
    s.base_cpu_frac = {
        Priority.QUERY: 0.12,
        Priority.INTERACTIVE: 0.08,
        Priority.BATCH_PIPELINE: 0.05,
    }
    s.base_ram_frac = {
        Priority.QUERY: 0.08,
        Priority.INTERACTIVE: 0.06,
        Priority.BATCH_PIPELINE: 0.04,
    }
    s.init_cpu_cap = {
        Priority.QUERY: 16.0,
        Priority.INTERACTIVE: 12.0,
        Priority.BATCH_PIPELINE: 8.0,
    }
    s.init_ram_cap = {
        Priority.QUERY: 192.0,
        Priority.INTERACTIVE: 160.0,
        Priority.BATCH_PIPELINE: 128.0,
    }

    # Per-tick scheduling controls
    s.max_assignments_per_tick = 256
    s.max_ops_per_pipeline_per_tick = {
        Priority.QUERY: 2,
        Priority.INTERACTIVE: 2,
        Priority.BATCH_PIPELINE: 1,
    }

    # When high-priority backlog exists, leave a little headroom (small, dynamic, work-conserving)
    s.reserve_cpu_frac = 0.06
    s.reserve_ram_frac = 0.06

    # Resource share caps per tick when multiple classes have backlog (prevents total starvation)
    s.share_cpu_caps = {
        Priority.QUERY: 0.30,
        Priority.INTERACTIVE: 0.50,
        Priority.BATCH_PIPELINE: 1.00,  # batch uses remainder
    }
    s.share_ram_caps = {
        Priority.QUERY: 0.30,
        Priority.INTERACTIVE: 0.50,
        Priority.BATCH_PIPELINE: 1.00,
    }

    # Batch aging: after long wait, allow higher per-op CPU to reduce timeout risk
    s.batch_aging_ticks = 80


@register_scheduler(key="scheduler_high_006")
def scheduler_high_006_scheduler(s, results, pipelines):
    s.tick += 1

    def _q_and_set(pri):
        if pri == Priority.QUERY:
            return s.q_query, s.enqueued_query
        if pri == Priority.INTERACTIVE:
            return s.q_interactive, s.enqueued_interactive
        return s.q_batch, s.enqueued_batch

    def _ensure_meta(pid):
        if pid not in s.pipeline_meta:
            s.pipeline_meta[pid] = {"arrival_tick": s.tick, "completed_ops": 0, "failures": 0}

    def _enqueue_pipeline(p):
        pid = p.pipeline_id
        s.pipeline_by_id[pid] = p
        _ensure_meta(pid)
        q, enq = _q_and_set(p.priority)
        if pid in enq:
            return
        q.append(pid)
        enq.add(pid)

    def _dequeue_pipeline_id(pid, pri):
        # Lazy removal is fine; this is for when we decide to stop considering it.
        q, enq = _q_and_set(pri)
        if pid in enq:
            enq.remove(pid)

    def _err_text(err):
        if err is None:
            return ""
        try:
            # dict-like errors sometimes carry structured info
            if isinstance(err, dict):
                parts = []
                for k, v in err.items():
                    try:
                        parts.append(str(k))
                        parts.append(str(v))
                    except Exception:
                        continue
                return " ".join(parts).lower()
        except Exception:
            pass
        try:
            name = ""
            try:
                name = err.__class__.__name__
            except Exception:
                name = ""
            return (name + " " + str(err)).lower()
        except Exception:
            try:
                return str(err).lower()
            except Exception:
                return "error"

    def _is_oom(err):
        t = _err_text(err)
        if not t:
            return False
        return ("oom" in t) or ("out of memory" in t) or ("memoryerror" in t) or ("killed" in t and "memory" in t)

    def _is_timeout(err):
        t = _err_text(err)
        if not t:
            return False
        return ("timeout" in t) or ("timed out" in t) or ("deadline" in t) or ("time limit" in t) or ("exceeded time" in t)

    def _is_non_retryable(err):
        # Be conservative: most failures are worth retrying in this simulator.
        t = _err_text(err)
        if not t:
            return False
        return ("syntax" in t) or ("parse" in t) or ("permission" in t) or ("auth" in t) or ("not found" in t)

    def _pool_max(pool_id):
        pool = s.executor.pools[pool_id]
        return float(pool.max_cpu_pool), float(pool.max_ram_pool)

    def _pool_avail(pool_id, avail_cpu, avail_ram):
        return float(avail_cpu[pool_id]), float(avail_ram[pool_id])

    def _init_guess(pri, pool_id):
        max_cpu, max_ram = _pool_max(pool_id)
        cpu = max(s.min_cpu[pri], s.base_cpu_abs[pri], max_cpu * s.base_cpu_frac[pri])
        ram = max(s.min_ram[pri], s.base_ram_abs[pri], max_ram * s.base_ram_frac[pri])
        cpu = min(cpu, s.init_cpu_cap[pri], max_cpu)
        ram = min(ram, s.init_ram_cap[pri], max_ram)
        return cpu, ram

    def _pipeline_age(pid):
        meta = s.pipeline_meta.get(pid)
        if not meta:
            return 0
        return s.tick - int(meta.get("arrival_tick", s.tick))

    def _pipeline_priority(pid):
        p = s.pipeline_by_id.get(pid)
        return getattr(p, "priority", Priority.BATCH_PIPELINE)

    def _has_runnable(q, scan=48):
        n = 0
        for pid in q:
            p = s.pipeline_by_id.get(pid)
            if p is None:
                continue
            if pid in s.pipeline_permanent_fail:
                continue
            st = p.runtime_status()
            if st.is_pipeline_successful():
                continue
            ops = st.get_ops(ASSIGNABLE_STATES, require_parents_complete=True)
            if ops:
                return True
            n += 1
            if n >= scan:
                break
        return False

    def _alloc_for_op(pool_id, pri, pid, op, avail_cpu_pool, avail_ram_pool, reserve_active):
        max_cpu, max_ram = _pool_max(pool_id)
        op_id = id(op)

        # Desired estimates (learned overrides initial guess)
        base_cpu, base_ram = _init_guess(pri, pool_id)
        est_cpu = s.op_cpu_est.get(op_id, base_cpu)
        est_ram = s.op_ram_est.get(op_id, base_ram)

        # Batch aging: if it's been waiting a long time, give it a bit more CPU to reduce timeout risk
        if pri == Priority.BATCH_PIPELINE and _pipeline_age(pid) >= s.batch_aging_ticks:
            est_cpu = min(max_cpu, max(est_cpu, base_cpu * 1.5))

        # If we've seen timeouts, treat CPU estimate as more "sticky"
        if s.op_timeouts.get(op_id, 0) > 0:
            est_cpu = max(est_cpu, base_cpu)

        # If we've seen OOMs, treat RAM estimate as hard-ish minimum
        oom_count = s.op_ooms.get(op_id, 0)
        hard_ram = (oom_count > 0)

        # Clamp to pool max
        est_cpu = max(s.min_cpu[pri], min(float(est_cpu), max_cpu))
        est_ram = max(s.min_ram[pri], min(float(est_ram), max_ram))

        # CPU: always shrink-to-fit if needed (CPU doesn't cause OOM)
        cpu = min(est_cpu, avail_cpu_pool)

        # RAM: if we have OOM history, don't under-allocate; otherwise allow shrink-to-fit
        if hard_ram:
            if avail_ram_pool + 1e-9 < est_ram:
                return None, None
            ram = est_ram
        else:
            # Shrink-to-fit for guessed RAM (keeps system work-conserving & improves throughput)
            ram = min(est_ram, avail_ram_pool)

        cpu = max(0.0, cpu)
        ram = max(0.0, ram)

        if cpu + 1e-9 < s.min_cpu[pri] or ram + 1e-9 < s.min_ram[pri]:
            return None, None

        # If reserve is active (protect query/interactive), ensure batch leaves headroom
        if pri == Priority.BATCH_PIPELINE and reserve_active:
            reserve_cpu = max_cpu * s.reserve_cpu_frac
            reserve_ram = max_ram * s.reserve_ram_frac
            if (avail_cpu_pool - cpu) + 1e-9 < reserve_cpu:
                return None, None
            if (avail_ram_pool - ram) + 1e-9 < reserve_ram:
                return None, None

        return cpu, ram

    def _best_pool_for(op, pri, pid, avail_cpu, avail_ram, reserve_active):
        best = None
        best_score = None
        for pool_id in range(s.executor.num_pools):
            a_cpu, a_ram = _pool_avail(pool_id, avail_cpu, avail_ram)
            if a_cpu <= 0.0 or a_ram <= 0.0:
                continue
            cpu, ram = _alloc_for_op(pool_id, pri, pid, op, a_cpu, a_ram, reserve_active)
            if cpu is None or ram is None:
                continue

            max_cpu, max_ram = _pool_max(pool_id)
            # Best-fit score (prefer tighter fits to reduce fragmentation)
            rem_cpu = max(0.0, a_cpu - cpu)
            rem_ram = max(0.0, a_ram - ram)
            score = (rem_cpu / (max_cpu + 1e-9)) ** 2 + (rem_ram / (max_ram + 1e-9)) ** 2

            if best is None or score < best_score:
                best = (pool_id, cpu, ram)
                best_score = score

        return best

    # Enqueue newly seen pipelines (and refresh references)
    for p in pipelines:
        s.pipeline_by_id[p.pipeline_id] = p
        _ensure_meta(p.pipeline_id)
        _enqueue_pipeline(p)

    # Learn from results
    for r in results:
        ops = getattr(r, "ops", None)
        if not ops:
            continue

        failed = False
        try:
            failed = bool(r.failed())
        except Exception:
            failed = bool(getattr(r, "error", None))

        # Attribute pipeline failure/completions
        pid = None
        # Try to resolve pipeline_id via any op
        for op in ops:
            pid = s.op_to_pipeline.get(id(op))
            if pid is not None:
                break

        if not failed:
            if pid is not None:
                meta = s.pipeline_meta.get(pid)
                if meta is not None:
                    try:
                        meta["completed_ops"] = int(meta.get("completed_ops", 0)) + len(ops)
                    except Exception:
                        meta["completed_ops"] = int(meta.get("completed_ops", 0)) + 1
            continue

        err = getattr(r, "error", None)
        oom = _is_oom(err)
        timeout = _is_timeout(err)
        non_retryable = _is_non_retryable(err)

        if pid is not None:
            meta = s.pipeline_meta.get(pid)
            if meta is not None:
                meta["failures"] = int(meta.get("failures", 0)) + 1

        for op in ops:
            op_id = id(op)
            s.op_failures[op_id] = int(s.op_failures.get(op_id, 0)) + 1

            pool_id = getattr(r, "pool_id", 0)
            try:
                max_cpu, max_ram = _pool_max(pool_id)
            except Exception:
                max_cpu, max_ram = 1.0, 1.0

            attempted_ram = float(getattr(r, "ram", 0.0) or 0.0)
            attempted_cpu = float(getattr(r, "cpu", 0.0) or 0.0)

            # Ensure some baseline even if result doesn't carry resource values
            if attempted_ram <= 0.0:
                attempted_ram = min(max_ram, max(1.0, s.op_ram_est.get(op_id, 1.0)))
            if attempted_cpu <= 0.0:
                attempted_cpu = min(max_cpu, max(0.5, s.op_cpu_est.get(op_id, 0.5)))

            if oom:
                s.op_ooms[op_id] = int(s.op_ooms.get(op_id, 0)) + 1
                new_ram = min(max_ram, max(s.op_ram_est.get(op_id, 0.0), attempted_ram) * 2.0)
                # Small additive bump helps escape float edge cases
                new_ram = min(max_ram, new_ram + 1e-6)
                s.op_ram_est[op_id] = max(s.min_ram[_pipeline_priority(pid)] if pid is not None else 0.0, new_ram)
            elif timeout:
                s.op_timeouts[op_id] = int(s.op_timeouts.get(op_id, 0)) + 1
                new_cpu = min(max_cpu, max(s.op_cpu_est.get(op_id, 0.0), attempted_cpu) * 1.6)
                new_cpu = min(max_cpu, new_cpu + 1e-6)
                s.op_cpu_est[op_id] = max(0.25, new_cpu)
                # Slight RAM bump can reduce spill-related slowdowns if modeled
                s.op_ram_est[op_id] = min(max_ram, max(s.op_ram_est.get(op_id, 0.0), attempted_ram) * 1.15)
            else:
                # Unknown failure: retry a few times with modest bumps (avoid premature permanent-fail)
                bump_ram = 1.25 if not non_retryable else 1.0
                bump_cpu = 1.25 if not non_retryable else 1.0
                s.op_ram_est[op_id] = min(max_ram, max(s.op_ram_est.get(op_id, 0.0), attempted_ram) * bump_ram + 0.5)
                s.op_cpu_est[op_id] = min(max_cpu, max(s.op_cpu_est.get(op_id, 0.0), attempted_cpu) * bump_cpu + 0.1)

        # Decide if we should permanently fail a pipeline (only when it's clearly hopeless)
        if pid is not None:
            meta = s.pipeline_meta.get(pid, {})
            too_many_pipeline_failures = int(meta.get("failures", 0)) >= s.max_pipeline_failures
            if too_many_pipeline_failures:
                s.pipeline_permanent_fail.add(pid)
            else:
                # If non-retryable error, fail fast
                if non_retryable:
                    s.pipeline_permanent_fail.add(pid)
                else:
                    # If an op OOM'd while already near max RAM, it's likely unschedulable
                    if oom:
                        try:
                            pool_id = getattr(r, "pool_id", 0)
                            _, max_ram = _pool_max(pool_id)
                            if attempted_ram >= 0.95 * float(max_ram):
                                s.pipeline_permanent_fail.add(pid)
                        except Exception:
                            pass

    suspensions = []
    assignments = []

    num_pools = s.executor.num_pools
    avail_cpu = [float(s.executor.pools[i].avail_cpu_pool) for i in range(num_pools)]
    avail_ram = [float(s.executor.pools[i].avail_ram_pool) for i in range(num_pools)]

    total_cpu0 = sum(avail_cpu)
    total_ram0 = sum(avail_ram)

    # Backlog signals (approximate) to drive reserve + share caps
    query_backlog = _has_runnable(s.q_query)
    interactive_backlog = _has_runnable(s.q_interactive)
    batch_backlog = _has_runnable(s.q_batch)
    high_backlog = query_backlog or interactive_backlog
    reserve_active_for_batch = high_backlog

    cpu_scheduled = {Priority.QUERY: 0.0, Priority.INTERACTIVE: 0.0, Priority.BATCH_PIPELINE: 0.0}
    ram_scheduled = {Priority.QUERY: 0.0, Priority.INTERACTIVE: 0.0, Priority.BATCH_PIPELINE: 0.0}

    # Dynamic caps: only enforce when multiple classes actually have runnable work
    enforce_caps = (interactive_backlog or batch_backlog)

    def _caps_for(pri):
        if not enforce_caps:
            return float("inf"), float("inf")
        if pri == Priority.BATCH_PIPELINE:
            return float("inf"), float("inf")
        cpu_cap = total_cpu0 * float(s.share_cpu_caps.get(pri, 1.0))
        ram_cap = total_ram0 * float(s.share_ram_caps.get(pri, 1.0))
        return cpu_cap, ram_cap

    per_pipeline_scheduled = {}  # pid -> count this tick

    def _pick_next_pipeline(q, pri, scan_limit=256):
        scanned = 0
        while q and scanned < scan_limit:
            pid = q.popleft()
            scanned += 1

            p = s.pipeline_by_id.get(pid)
            if p is None:
                _dequeue_pipeline_id(pid, pri)
                continue

            if pid in s.pipeline_permanent_fail:
                _dequeue_pipeline_id(pid, pri)
                continue

            st = p.runtime_status()
            if st.is_pipeline_successful():
                _dequeue_pipeline_id(pid, pri)
                continue

            # Keep it in rotation
            q.append(pid)

            # If we've exceeded retry bounds for all failing ops, mark permanent fail
            # (soft guard; actual per-op learning handled in results)
            # We do not inspect ops here to keep it cheap.

            # Find runnable ops
            ops = st.get_ops(ASSIGNABLE_STATES, require_parents_complete=True)
            if not ops:
                continue

            # Prefer finishing pipelines already in-progress (reduces penalty-heavy incompletes)
            meta = s.pipeline_meta.get(pid, {})
            in_progress = int(meta.get("completed_ops", 0)) > 0
            if in_progress:
                return pid, p, ops
            # If not in progress, still acceptable; return if nothing better found quickly
            return pid, p, ops

        return None, None, None

    def _schedule_priority(pri, q):
        nonlocal assignments, avail_cpu, avail_ram

        cpu_cap, ram_cap = _caps_for(pri)
        no_fit_streak = 0

        for _ in range(s.max_assignments_per_tick):
            if sum(avail_cpu) <= 1e-9 or sum(avail_ram) <= 1e-9:
                return

            if cpu_scheduled[pri] >= cpu_cap + 1e-9 or ram_scheduled[pri] >= ram_cap + 1e-9:
                return

            pid, p, ops = _pick_next_pipeline(q, pri)
            if pid is None:
                return

            # Per-pipeline per-tick limit
            lim = int(s.max_ops_per_pipeline_per_tick.get(pri, 1))
            already = int(per_pipeline_scheduled.get(pid, 0))
            if already >= lim:
                continue

            # Schedule up to remaining quota
            remaining = lim - already
            ops_to_try = ops[:remaining]
            placed_any = False

            for op in ops_to_try:
                # Stop if we hit caps mid-pipeline
                if cpu_scheduled[pri] >= cpu_cap + 1e-9 or ram_scheduled[pri] >= ram_cap + 1e-9:
                    break

                choice = _best_pool_for(op, pri, pid, avail_cpu, avail_ram, reserve_active_for_batch)
                if choice is None:
                    continue

                pool_id, cpu, ram = choice

                # Also respect caps by shrinking CPU if needed (work-conserving)
                cpu_room = cpu_cap - cpu_scheduled[pri]
                ram_room = ram_cap - ram_scheduled[pri]
                if enforce_caps and pri != Priority.BATCH_PIPELINE:
                    if cpu > cpu_room + 1e-9:
                        cpu = max(s.min_cpu[pri], min(cpu, cpu_room))
                    if ram > ram_room + 1e-9:
                        # Only shrink guessed RAM; learned hard-min (OOM history) is enforced in allocator
                        ram = max(s.min_ram[pri], min(ram, ram_room))

                    if cpu + 1e-9 < s.min_cpu[pri] or ram + 1e-9 < s.min_ram[pri]:
                        continue

                    # Re-check fit after shrinking
                    if cpu > avail_cpu[pool_id] + 1e-9 or ram > avail_ram[pool_id] + 1e-9:
                        continue

                    # If batch reserve is active, ensure shrinking didn't violate reserve constraints for batch
                    if pri == Priority.BATCH_PIPELINE and reserve_active_for_batch:
                        max_cpu, max_ram = _pool_max(pool_id)
                        reserve_cpu = max_cpu * s.reserve_cpu_frac
                        reserve_ram = max_ram * s.reserve_ram_frac
                        if (avail_cpu[pool_id] - cpu) + 1e-9 < reserve_cpu:
                            continue
                        if (avail_ram[pool_id] - ram) + 1e-9 < reserve_ram:
                            continue

                assignments.append(
                    Assignment(
                        ops=[op],
                        cpu=cpu,
                        ram=ram,
                        priority=p.priority,
                        pool_id=pool_id,
                        pipeline_id=pid,
                    )
                )
                s.op_to_pipeline[id(op)] = pid

                avail_cpu[pool_id] = max(0.0, avail_cpu[pool_id] - cpu)
                avail_ram[pool_id] = max(0.0, avail_ram[pool_id] - ram)
                cpu_scheduled[pri] += cpu
                ram_scheduled[pri] += ram
                per_pipeline_scheduled[pid] = int(per_pipeline_scheduled.get(pid, 0)) + 1

                placed_any = True
                no_fit_streak = 0

            if not placed_any:
                no_fit_streak += 1
                if no_fit_streak >= 24:
                    return

    # Phase 1: Query (bounded share when others exist)
    _schedule_priority(Priority.QUERY, s.q_query)

    # Phase 2: Interactive
    _schedule_priority(Priority.INTERACTIVE, s.q_interactive)

    # Phase 3: Batch
    _schedule_priority(Priority.BATCH_PIPELINE, s.q_batch)

    # Work-conserving backfill: if we left headroom due to reserve but no high backlog exists now, fill with batch
    if (not high_backlog) and batch_backlog and (sum(avail_cpu) > 1e-6 and sum(avail_ram) > 1e-6):
        reserve_active_for_batch = False
        _schedule_priority(Priority.BATCH_PIPELINE, s.q_batch)

    return suspensions, assignments
